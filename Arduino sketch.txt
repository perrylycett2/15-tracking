#include <Arduino.h>
#include <SPI.h>
#include <MFRC522.h>
 
#define DEBUG
 
const byte NUM_READERS = 2;
const byte SS_PINS[NUM_READERS] = {5, 17};
const byte RST_PIN = 21;
 
// ESP32 VSPI pins
const byte SPI_SCK  = 18;
const byte SPI_MISO = 19;
const byte SPI_MOSI = 23;
 
MFRC522 readers[NUM_READERS] = {
  MFRC522(SS_PINS[0], RST_PIN),
  MFRC522(SS_PINS[1], RST_PIN)
};
 
String lastUID[NUM_READERS] = {"", ""};
unsigned long lastReadTime[NUM_READERS] = {0, 0};
 
String uidToString(MFRC522::Uid *uid) {
  String out = "";
  for (byte i = 0; i < uid->size; i++) {
    if (uid->uidByte[i] < 0x10) out += "0";
    out += String(uid->uidByte[i], HEX);
    if (i < uid->size - 1) out += ":";
  }
  out.toUpperCase();
  return out;
}
 
void deselectAllReaders() {
  for (byte i = 0; i < NUM_READERS; i++) {
    digitalWrite(SS_PINS[i], HIGH);
  }
}
 
void initReader(byte i) {
  deselectAllReaders();
  digitalWrite(SS_PINS[i], LOW);
  delay(5);
 
  readers[i].PCD_Init();
  delay(20);
  readers[i].PCD_AntennaOn();
 
#ifdef DEBUG
  Serial.print("Reader ");
  Serial.print(i + 1);
  Serial.print(" on SS pin ");
  Serial.print(SS_PINS[i]);
  Serial.print(" | Version: ");
  readers[i].PCD_DumpVersionToSerial();
#endif
 
  digitalWrite(SS_PINS[i], HIGH);
}
 
bool scanReader(byte i) {
  deselectAllReaders();
  digitalWrite(SS_PINS[i], LOW);
  delayMicroseconds(50);
 
  if (!readers[i].PICC_IsNewCardPresent()) {
    digitalWrite(SS_PINS[i], HIGH);
    return false;
  }
 
  if (!readers[i].PICC_ReadCardSerial()) {
    digitalWrite(SS_PINS[i], HIGH);
    return false;
  }
 
  String uid = uidToString(&readers[i].uid);
 
  // simple anti-spam debounce
  if (uid != lastUID[i] || millis() - lastReadTime[i] > 1000) {
    Serial.print("Reader ");
    Serial.print(i + 1);
    Serial.print(" scanned card UID: ");
    Serial.println(uid);
 
    lastUID[i] = uid;
    lastReadTime[i] = millis();
  }
 
  readers[i].PICC_HaltA();
  readers[i].PCD_StopCrypto1();
  digitalWrite(SS_PINS[i], HIGH);
  return true;
}
 
void setup() {
#ifdef DEBUG
  Serial.begin(115200);
  delay(1000);
  Serial.println("\nStarting dual RC522 test...");
#endif
 
  pinMode(RST_PIN, OUTPUT);
  digitalWrite(RST_PIN, HIGH);
 
  for (byte i = 0; i < NUM_READERS; i++) {
    pinMode(SS_PINS[i], OUTPUT);
    digitalWrite(SS_PINS[i], HIGH);
  }
 
  SPI.begin(SPI_SCK, SPI_MISO, SPI_MOSI);
 
  for (byte i = 0; i < NUM_READERS; i++) {
    initReader(i);
  }
 
#ifdef DEBUG
  Serial.println("Ready. Tap a card on either reader.");
#endif
}
 
void loop() {
  for (byte i = 0; i < NUM_READERS; i++) {
    scanReader(i);
    delay(10);
  }
}
